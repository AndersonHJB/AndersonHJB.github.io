    func = sqrt(x) * ln(x)
    return func.subs(x, x_val).evalf()

# 定义导数函数 fp(x)
def fp(x_val):
    derivative = (ln(x)/(2*sqrt(x))) + (1/sqrt(x))
    return derivative.subs(x, x_val).evalf()

def fp_approx(f, a, h):
