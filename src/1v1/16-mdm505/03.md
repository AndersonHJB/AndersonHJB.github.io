```python
def sun_fun(a, b, c):
    if c == '+':
        return a + b
    elif c == '-':
        return a - b
    elif c == "*":
        return a * b
    elif c == "/":
        return a / b


def loop():
    while True:
        target = input(">>")
        if target.lower() == "q":
            break
        targets = target.split(",")
        a, b = map(int, targets[:2])
        c = targets[2]
        print(a, b, c)
        print(sun_fun(a, b, c))


if __name__ == '__main__':
    loop()
```
